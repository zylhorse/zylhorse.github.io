生成自签名SSL免密证书

2015-06-13

zylhorse

5分钟 

#!/bin/sh
  
# create self-signed server certificate:

SSLDIR=$PWD/ssl
SERVERDIR=$SSLDIR/server
CLIENTDIR=$SSLDIR/client
OPENSSLFILE=$PWD/openssl.cnf

sudo rm -rf $SSLDIR
sudo mkdir $SSLDIR
sudo mkdir $SERVERDIR
sudo mkdir $CLIENTDIR
sudo mkdir $SSLDIR/demoCA
sudo mkdir $SSLDIR/demoCA/newcerts
sudo touch $SSLDIR/demoCA/index.txt
sudo touch $SSLDIR/demoCA/serial
sudo echo 01 > $SSLDIR/demoCA/serial
cd $SSLDIR

read -p "Enter your domain [www.example.com]: " DOMAIN
sudo sed -i "s/subjectAltName=IP.*/subjectAltName=IP:$DOMAIN/g" $OPENSSLFILE

SUBJECT="/C=CN/ST=Beijing/L=Beigjing/O=YongNuo/OU=CEO/CN=$DOMAIN/emailAddress=zylhorse@126.com"

echo "Create Ca key / cert ..."
sudo openssl req -new -x509 -subj $SUBJECT -keyout $SSLDIR/ca.key -out $SSLDIR/ca.crt -days 365 -nodes
sudo openssl rsa -in $SSLDIR/ca.key -out $SSLDIR/ca.key

echo "Create server key..."
sudo openssl genrsa -des3 -passout pass:123456 -out $SERVERDIR/$DOMAIN.key 1024
sudo openssl rsa -passin pass:123456 -in $SERVERDIR/$DOMAIN.key -passout pass:123456 -out $SERVERDIR/$DOMAIN.key

echo "Create server certificate signing request..."
sudo openssl req -new -subj $SUBJECT -key $SERVERDIR/$DOMAIN.key -out $SERVERDIR/$DOMAIN.csr -config $OPENSSLFILE -extensions v3_req

echo "Sign SSL certificate..."
sudo openssl ca -passin pass:123456 -in $SERVERDIR/$DOMAIN.csr -out $SERVERDIR/$DOMAIN.crt -cert $SSLDIR/ca.crt -keyfile $SSLDIR/ca.key -config $OPENSSLFILE -extensions v3_req

echo "Create client key..."
sudo openssl genrsa -des3 -passout pass:123456 -out $CLIENTDIR/$DOMAIN.key 1024
sudo openssl rsa -passin pass:123456 -in $CLIENTDIR/$DOMAIN.key -passout pass:123456 -out $CLIENTDIR/$DOMAIN.key

echo "Create client certificate signing request..."
sudo openssl req -new -subj $SUBJECT -key $CLIENTDIR/$DOMAIN.key -out $CLIENTDIR/$DOMAIN.csr -config $OPENSSLFILE -extensions v3_req

echo "Sign SSL certificate..."
sudo openssl ca -passin pass:123456 -in $CLIENTDIR/$DOMAIN.csr -out $CLIENTDIR/$DOMAIN.crt -cert $SSLDIR/ca.crt -keyfile $SSLDIR/ca.key -config $OPENSSLFILE -extensions v3_req
