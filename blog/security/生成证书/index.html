生成自签名SSL证书

2015-06-22

zylhorse

5分钟 

#!/bin/sh

# create self-signed server certificate:

SSLDIR=$PWD/ssl
SERVERDIR=$SSLDIR/server
CLIENTDIR=$SSLDIR/client
OPENSSLFILE=$PWD/openssl.cnf

su root

sudo rm -rf $SSLDIR
sudo mkdir $SSLDIR
sudo mkdir $SERVERDIR
sudo mkdir $CLIENTDIR
sudo mkdir $SSLDIR/demoCA
sudo mkdir $SSLDIR/demoCA/newcerts
sudo touch $SSLDIR/demoCA/index.txt
sudo touch $SSLDIR/demoCA/serial
sudo echo 01 > $SSLDIR/demoCA/serial
cd $SSLDIR

read -p "Enter your domain [www.example.com]: " DOMAIN
sudo sed -e "s|192.168.10.33|$DOMAIN|g" -i $OPENSSLFILE

SUBJECT="/C=CN/ST=Beijing/L=Beigjing/O=YongNuo/OU=CEO/CN=$DOMAIN/emailAddress=zylhorse@126.com"

echo "Create Ca key / cert ..."
sudo openssl req -new -x509 -subj $SUBJECT -keyout $SSLDIR/ca.key -out $SSLDIR/ca.crt -days 365
sudo openssl rsa -in $SSLDIR/ca.key -out $SSLDIR/ca.key

echo "Create server key..."
sudo openssl genrsa -des3 -out $SERVERDIR/$DOMAIN.key 1024
sudo openssl rsa -in $SERVERDIR/$DOMAIN.key -out $SERVERDIR/$DOMAIN.key

echo "Create server certificate signing request..."
sudo openssl req -new -subj $SUBJECT -key $SERVERDIR/$DOMAIN.key -out $SERVERDIR/$DOMAIN.csr -config $OPENSSLFILE -extensions v3_req

echo "Sign SSL certificate..."
sudo openssl ca -in $SERVERDIR/$DOMAIN.csr -out $SERVERDIR/$DOMAIN.crt -cert $SSLDIR/ca.crt -keyfile $SSLDIR/ca.key -config $OPENSSLFILE -extensions v3_req

echo "Create client key..."
sudo openssl genrsa -des3 -out $CLIENTDIR/$DOMAIN.key 1024
sudo openssl rsa -in $CLIENTDIR/$DOMAIN.key -out $CLIENTDIR/$DOMAIN.key

echo "Create client certificate signing request..."
sudo openssl req -new -subj $SUBJECT -key $CLIENTDIR/$DOMAIN.key -out $CLIENTDIR/$DOMAIN.csr -config $OPENSSLFILE -extensions v3_req

echo "Sign SSL certificate..."
sudo openssl ca -in $CLIENTDIR/$DOMAIN.csr -out $CLIENTDIR/$DOMAIN.crt -cert $SSLDIR/ca.crt -keyfile $SSLDIR/ca.key -config $OPENSSLFILE -extensions v3_req


echo "copy crt/key -> /etc/nginx/ssl/" 
sudo cp $SSLDIR/ca.crt $SERVERDIR/$DOMAIN.crt $SERVERDIR/$DOMAIN.key /etc/nginx/ssl

echo "import ca.crt to tomcat"
cd /usr/lib/jvm/java-8-openjdk-amd64/jre/lib/security
sudo keytool -keystore cacerts -delete -alias $DOMAIN -storepass changeit
sudo keytool -keystore cacerts -importcert -alias $DOMAIN -file /etc/nginx/ssl/ca.crt -storepass changeit
